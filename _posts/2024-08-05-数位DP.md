## 会用到的思想

1. 前缀和
2. 树结构

对于范围内的数，`N`:

$$
N = a_{n-1}a_{n-2}...a_0
$$

![](assets/img/2024-08-05-数位DP/think.png)

也就是所有叶子结点的组合数的和。

而计算组合数也可以通过递归得到：

$$C(a,b) = C(a - 1, b) + C(a - 1, b - 1)$$


## 通用代码实现

以该题为模板修改即可：[模板](https://www.acwing.com/problem/content/1083/)

```c++
#include <cstring>
#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

const int N = 35;

int K, B;


// 如果碰到同类型的题，需要修改的内容：记忆化数组 f 以及原始实现范围内容
// C(a, b)
int f[N][N];

void init(){
    for(int i = 0; i < N; i++){
        for(int j = 0; j <= i; j++){
            if(!j) f[i][j] = 1;
            else f[i][j] = f[i - 1][j - 1] + f[i - 1][j];
        }
    }
}

int dp(int n){
    if(!n) return 0;
    vector<int> num;
    while(n) num.emplace_back(n % B), n /= B;

    int res = 0;
    int last = 0;
    for(int i = num.size() - 1; i >= 0; i--){
        int x =  num[i];
        // 原始实现
        if(x){
            // 左边分支
            res += f[i][K - last];
            if(x > 1){
                if( K - last - 1 >= 0 )res += f[i][K - last - 1];
                break;
            }
            else{
                last++;
                if(last > K) break;
            }
        }

        if(!i && last == K) res++;
        // 原始实现
    }

    return res;
}

int main(){
    init();
    
    int l, r;
    cin >> l >> r >> K >> B;

    cout << dp(r) - dp(l - 1) << '\n';

    return 0;
}
```
